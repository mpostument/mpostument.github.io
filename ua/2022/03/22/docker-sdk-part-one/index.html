Docker Sdk Part One

2022-03-22 :: Maksym Postument :: 3 min read (455 words)
#Go  #Docker  #Beginner 

Привіт!

Це перша частина статей про docker sdk. В ній розглянемо як за допомогою golang можна робити pull та list docker images.

Docker pull⌗

Як зазвичай ініціалізуємо go модуль, завантажимо docker sdk і створимо файл main.go

go mod init
go get github.com/docker/docker/client

Створимо метод PullImage в якому створимо docker client за допомогою методу client.NewClientWithOpts, як аргументи до цього методу передамо client.FromEnv який візьме конфігурацію docker зі змінних оточення і client.WithAPIVersionNegotiation() який автоматично визначить доступну версію docker API. Тепер за допомогою клієнта і методу ImagePull можна завантажити потрібний docker image. В метод ImagePull потрібно передати контекст, назву docker image який можна завантажити, і types.ImagePullOptions{}. В ImagePullOptions можна передати дані авторизації якщо вони потрібні для завантаження docker image. В самому кінці викличемо io.Copy(os.Stdout, out) для того, щоб вивести результат завантаження. У функції main викличемо PullImage для того, щоб завантажити docker image.

package main

import (
	"context"
	"io"
	"os"
    "log"

	"github.com/docker/docker/api/types"
	"github.com/docker/docker/client"
)

func PullImage(imageName string) error {
	ctx := context.Background()
	dockerClient, err := client.NewClientWithOpts(client.FromEnv, client.WithAPIVersionNegotiation())

	if err != nil {
		return err
	}

	out, err := dockerClient.ImagePull(ctx, imageName, types.ImagePullOptions{})
	if err != nil {
		return err
	}
	defer out.Close()

	io.Copy(os.Stdout, out)
	return nil
}

func main() {
	err := PullImage("debian")
	if err != nil {
		log.Fatalln(err)
	}
}

Docker list⌗

Тепер напишемо метод для виведення списку docker image. Для цього створимо метод ListAllImages. Як і в попередньому коді створимо docker клієнт. Далі потрібен фільтр який можна застосувати, щоб вивести docker image за якимось критеріями. Оскільки я хочу виводити всі docker images то і фільтр залишиться пустим. Ну і тепер викличемо метод ImageList в який потрібно передати контекст та структуру ImageListOptions яка має два поля All та Filters. В Filters передамо попередньо створений фільтр, а в All вкажемо false. Якщо поставити All в true, то будуть виведені також всі проміжні docker images. Тепер в циклі можна ввести інформацію по всіх контейнерах.

package main

import (
	"context"
	"fmt"

	"github.com/docker/docker/api/types"
	"github.com/docker/docker/api/types/filters"
	"github.com/docker/docker/client"
)

func ListAllImages() error {
	ctx := context.Background()
	dockerClient, err := client.NewClientWithOpts(client.FromEnv, client.WithAPIVersionNegotiation())

	if err != nil {
		return err
	}

	filter := filters.NewArgs()

	images, err := dockerClient.ImageList(ctx, types.ImageListOptions{
		All:     false,
		Filters: filter,
	})

	if err != nil {
		return err
	}

	for _, i := range images {
		fmt.Println(i)
	}
	return nil
}
